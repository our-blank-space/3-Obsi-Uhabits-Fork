import { Plugin, WorkspaceLeaf } from "obsidian";
import { HabitStorage } from "./src/core/storage";
import { HomeView, HABIT_VIEW_TYPE } from "./src/ui/views/HomeView";
import { HabitSettingsTab } from "./src/ui/settings";

export default class HabitTrackerPlugin extends Plugin {
	storage!: HabitStorage;

	async onload() {
		console.log("Cargando Habit Loop Tracker...");

		// 1. Cargar Storage
		this.storage = new HabitStorage(this);
		await this.storage.load();

		// 2. Registrar Vista
		this.registerView(
			HABIT_VIEW_TYPE,
			(leaf) => new HomeView(leaf, this.storage)
		);

		// 3. Ribbon
		this.addRibbonIcon("check-circle-2", "Habit Tracker", () => {
			this.activateView();
		});

		// 4. Comandos
		this.addCommand({
			id: "open-habit-tracker",
			name: "Abrir Habit Tracker",
			callback: () => this.activateView()
		});

		// 5. Configuración
		this.addSettingTab(new HabitSettingsTab(this.app, this, this.storage));
	}

	async onunload() {
		this.app.workspace.detachLeavesOfType(HABIT_VIEW_TYPE);
	}

	async activateView() {
		const { workspace } = this.app;
		let leaf: WorkspaceLeaf | null = null;
		const leaves = workspace.getLeavesOfType(HABIT_VIEW_TYPE);

		if (leaves.length > 0) {
			leaf = leaves[0];
		} else {
			leaf = workspace.getRightLeaf(false);
			if (leaf) await leaf.setViewState({ type: HABIT_VIEW_TYPE, active: true });
		}

		if (leaf) workspace.revealLeaf(leaf);
	}
}

import { App, TFile, TFolder, normalizePath, Notice } from "obsidian";
import { Habit } from "../core/types";
import { toDateOnly } from "./dates";

export interface NoteSettings {
	folder: string;
	openAfterCreate: boolean;
	askBeforeCreate: boolean;
	filenamePattern: string;
	template: string;
}

export async function ensureFolder(app: App, folder: string): Promise<string> {
	const path = normalizePath(folder.trim() || "Habit Logs");
	const existing = app.vault.getAbstractFileByPath(path);
	if (!existing) {
		await app.vault.createFolder(path);
	}
	return path;
}

function renderTemplate(template: string, ctx: { habit: string; date: Date; value: string; notes: string; }) {
	let out = template;
	out = out.replace(/{{habit}}/g, ctx.habit);
	out = out.replace(/{{value}}/g, ctx.value);
	out = out.replace(/{{notes}}/g, ctx.notes);
	out = out.replace(/{{date:YYYY-MM-DD}}/g, () => {
		const y = ctx.date.getFullYear();
		const m = String(ctx.date.getMonth() + 1).padStart(2, "0");
		const d = String(ctx.date.getDate()).padStart(2, "0");
		return `${y}-${m}-${d}`;
	});
    // Helper simple de tiempo
    const now = new Date();
	out = out.replace(/{{time:HH:mm}}/g, `${String(now.getHours()).padStart(2,'0')}:${String(now.getMinutes()).padStart(2,'0')}`);
	return out;
}

export async function createHabitNote(
	app: App,
	settings: NoteSettings,
	habit: Habit,
	dateStr: string,
	valueText: string,
	userNotes: string
): Promise<string> {
	const baseFolder = await ensureFolder(app, settings.folder);
	const noteDate = toDateOnly(dateStr); 
	
	let filename = settings.filenamePattern.replace("{{habit}}", habit.name);
    const y = noteDate.getFullYear();
    const m = String(noteDate.getMonth() + 1).padStart(2, "0");
    const d = String(noteDate.getDate()).padStart(2, "0");
    filename = filename.replace("{{date:YYYY-MM-DD}}", `${y}-${m}-${d}`);
    
	if (!filename.endsWith(".md")) filename += ".md";
	const path = normalizePath(`${baseFolder}/${filename}`);
	
	const content = renderTemplate(settings.template, {
		habit: habit.name,
		date: noteDate,
		value: valueText,
		notes: userNotes
	});

	const existing = app.vault.getAbstractFileByPath(path);
	let file: TFile;

	try {
		if (existing instanceof TFile) {
			await app.vault.modify(existing, content);
			file = existing;
		} else {
			file = await app.vault.create(path, content);
		}
	} catch (e) {
		console.error(e);
		new Notice("Error creando nota.");
		throw e;
	}

	if (settings.openAfterCreate) {
		await app.workspace.getLeaf(true).openFile(file);
	}
	return file.path;
}

import { App, TFile, normalizePath } from "obsidian";
import { HabitStorage } from "../core/storage";

export async function exportJsonBackup(app: App, storage: HabitStorage): Promise<string> {
	const d = storage.getData();
	const folder = await ensureFolder(app, d.settingsSnapshot.backupFolder);
	const stamp = new Date().toISOString().replace(/[:.]/g, "-").slice(0, 19);
	const filename = `habit-backup-${stamp}.json`;
	const path = normalizePath(`${folder}/${filename}`);
	const json = JSON.stringify(d, null, 2);
	await app.vault.create(path, json);
	return path;
}

export async function importJsonBackup(app: App, storage: HabitStorage, path: string) {
	const normalized = normalizePath(path.trim());
	const file = app.vault.getAbstractFileByPath(normalized);
	if (!(file instanceof TFile)) throw new Error("Archivo no encontrado.");
	const content = await app.vault.read(file);
	const raw = JSON.parse(content);
	await storage.importFromRaw(raw);
}

export async function exportCsv(app: App, storage: HabitStorage): Promise<string> {
	const d = storage.getData();
	const folder = await ensureFolder(app, d.settingsSnapshot.backupFolder);
	const stamp = new Date().toISOString().replace(/[:.]/g, "-").slice(0, 19);
	const filename = `habit-logs-${stamp}.csv`;
	const path = normalizePath(`${folder}/${filename}`);

	const lines: string[] = [];
	lines.push("habitId,habitName,date,type,value,notePath");
	
	for (const h of d.habits) {
		for (const date of Object.keys(h.entries)) {
			const e = h.entries[date];
			let v = "";
			if (typeof e.value === "number") v = String(e.value);
			else v = e.value;
			const note = e.notePath ?? "";
			lines.push(`"${h.id}","${h.name.replace(/"/g, '""')}",${date},${h.type},${v},"${note.replace(/"/g, '""')}"`);
		}
	}
	await app.vault.create(path, lines.join("\n"));
	return path;
}

    await storage.repairData();
}

import { App, TFile, normalizePath } from "obsidian";
import { HabitStorage } from "../core/storage";
import { ensureFolder } from "./notes";
import { todayString, addDays, compareDateStr } from "./dates";
import { evalHabitOnDate } from "../core/entries";

export async function generateMonthlyReport(app: App, storage: HabitStorage): Promise<string> {
	const d = storage.getData();
	const folder = await ensureFolder(app, d.settingsSnapshot.reportsFolder);
	const now = new Date();
	const label = `${now.getFullYear()}-${String(now.getMonth() + 1).padStart(2, "0")}`;
	
	const filename = `Habit Report ${label}.md`;
	const path = normalizePath(`${folder}/${filename}`);
	let content = `# Habit Report (${label})\n\n`;

	for (const h of d.habits) {
		const startOfMonth = `${label}-01`;
		const nextMonth = new Date(now.getFullYear(), now.getMonth() + 1, 0); 
		const endOfMonth = `${label}-${String(nextMonth.getDate()).padStart(2, "0")}`;
		
		let ok = 0, total = 0;
		for (let d = startOfMonth; compareDateStr(d, endOfMonth) <= 0; d = addDays(d, 1)) {
			if (compareDateStr(d, todayString()) > 0) break;
			if (evalHabitOnDate(h, d) !== "NONE") {
                total++;
                if(evalHabitOnDate(h, d) === "OK") ok++;
            }
		}
		const percent = total > 0 ? Math.round((ok / total) * 100) : 0;
		content += `## ${h.name}\n- Score: ${ok}/${total} (${percent}%)\n\n`;
	}

	const existing = app.vault.getAbstractFileByPath(path);
	if (existing instanceof TFile) await app.vault.modify(existing, content);
	else await app.vault.create(path, content);
	return path;
}

===== src/utils/dates.ts =====

export function todayString(): string {
	const d = new Date();
	const y = d.getFullYear();
	const m = String(d.getMonth() + 1).padStart(2, "0");
	const day = String(d.getDate()).padStart(2, "0");
	return `${y}-${m}-${day}`;
}

export function toDateOnly(dateStr: string): Date {
	const [y, m, d] = dateStr.split("-").map((x) => Number(x));
	return new Date(Date.UTC(y, m - 1, d));
}

export function addDays(dateStr: string, offset: number): string {
	const d = toDateOnly(dateStr);
	d.setUTCDate(d.getUTCDate() + offset);
	const y = d.getUTCFullYear();
	const m = String(d.getUTCMonth() + 1).padStart(2, "0");
	const day = String(d.getUTCDate()).padStart(2, "0");
	return `${y}-${m}-${day}`;
}

export function compareDateStr(a: string, b: string): number {
	if (a < b) return -1;
	if (a > b) return 1;
	return 0;
}

export function weekdayShort(dateStr: string, firstDayOfWeek: "Mon" | "Sun" = "Mon"): string {
	const d = toDateOnly(dateStr);
	const day = d.getUTCDay(); 
	const names = ["Sun", "Mon", "Tue", "Wed", "Thu", "Fri", "Sat"];
	return names[day]; 
}

export function getRangeBounds(mode: "week" | "month" | "year"): { from: string; to: string } {
	const today = todayString();
	let from = today;

	if (mode === "week") {
		from = addDays(today, -6); 
	} else if (mode === "month") {
		from = addDays(today, -29); 
	} else {
		from = addDays(today, -364); 
	}

	return { from, to: today };
}

import { HabitStorage } from "./storage";
import { Habit, HabitEntry, HabitEval } from "./types";

/* ============================================
   Evaluación de entradas (sin cambios)
============================================ */

export function evalHabitEntry(habit: Habit, entry?: HabitEntry): HabitEval {
    if (!entry) return "NONE";

    if (habit.type === "yesno") {
        if (entry.value === "✔") return "OK";
        if (entry.value === "✖") return "NO";
        return "NONE";
    } else {
        if (typeof entry.value !== "number") return "NONE";
        if (!habit.goal) return "NONE";
        const v = entry.value;

        switch (habit.goal.type) {
            case "atLeast": return v >= habit.goal.value ? "OK" : "NO";
            case "atMost": return v <= habit.goal.value ? "OK" : "NO";
            case "exactly": return v === habit.goal.value ? "OK" : "NO";
            default: return "NONE";
        }
    }
}

export function evalHabitOnDate(habit: Habit, date: string): HabitEval {
    const entry = habit.entries[date];
    const basic = evalHabitEntry(habit, entry);
    if (basic !== "NONE") return basic;

    if (habit.type === "yesno") {
        const today = new Date().toISOString().split("T")[0];
        if (date < today) return "NO";
    }

    return basic;
}

   🔥 EXTENSIÓN: setEntry ahora soporta energía + ánimo

interface EntryExtraFields {
    energy?: number;
    mood?: string;
}

export async function setEntry(
    storage: HabitStorage,
    habitId: string,
    date: string,
    value: "✔" | "✖" | number,
    notePath?: string,
    extra?: EntryExtraFields
) {
    await storage.update((d) => {
        const h = d.habits.find((x) => x.id === habitId);
        if (!h) return;

        // Mantener datos previos si existen
        const prev = h.entries[date] ?? {};

        h.entries[date] = {
            date,
            value,
            notePath: notePath ?? prev.notePath,

            // 🔥 NUEVOS CAMPOS (con fallback)
            energy: extra?.energy ?? prev.energy,
            mood: extra?.mood ?? prev.mood
        };
    });
}

/* ============================================
   Helpers auxiliares
============================================ */

export async function clearEntry(storage: HabitStorage, habitId: string, date: string) {
    await storage.update((d) => {
        const h = d.habits.find((x) => x.id === habitId);
        if (!h) return;
        delete h.entries[date];
    });
}

export const deleteEntry = clearEntry;

export function getEntryForDate(
    storage: HabitStorage,
    habitId: string,
    date: string
): HabitEntry | undefined {
    const habit = storage.getData().habits.find(h => h.id === habitId);
    if (!habit) return undefined;
    return habit.entries[date];
}

import { Plugin, Events } from "obsidian";
import { HabitData, HabitPluginSettings, Habit } from "./types";

const CURRENT_DATA_VERSION = 1;

const DEFAULT_SETTINGS: HabitPluginSettings = {
	firstDayOfWeek: "Mon",
	notesFolder: "Habit Logs",
	openNoteAfterCreate: false,
	askBeforeCreateNote: false,
	noteFilenamePattern: "{{date:YYYY-MM-DD}} - {{habit}}",
	noteTemplate: "---\ntype: habit-log\n---\n# {{habit}}\n{{notes}}",
	reportsFolder: "Habit Logs",
	dayBarOrientation: "recent-right",
	autoHideCompletedToday: false,
	soundsEnabled: false,
	backupFolder: "Habit Backups",
	sortMode: "manual"
};

export class HabitStorage {
	private plugin: Plugin;
	private data: HabitData;
	
	public events = new Events();

	constructor(plugin: Plugin) {
		this.plugin = plugin;
		this.data = this.getEmptyData();
	}

	async load(): Promise<HabitData> {
		const raw = await this.plugin.loadData();
		if (!raw) {
			this.data = this.getEmptyData();
			await this.save();
			return this.data;
		}
		this.data = this.migrate(raw);
		return this.data;
	}

	async save(): Promise<void> {
		await this.plugin.saveData(this.data);
		this.events.trigger("changed", this.data);
	}

	getData(): HabitData {
		return this.data;
	}

	async update(mutator: (d: HabitData) => void): Promise<void> {
		mutator(this.data);
		await this.save();
	}

	private getEmptyData(): HabitData {
		return {
			version: CURRENT_DATA_VERSION,
			habits: [],
			settingsSnapshot: { ...DEFAULT_SETTINGS }
		};
	}

	private migrate(raw: any): HabitData {
		const v: any = raw ?? {};
		if (v.version == null) v.version = 1;
		if (!Array.isArray(v.habits)) v.habits = [];
		v.settingsSnapshot = { ...DEFAULT_SETTINGS, ...v.settingsSnapshot };
		v.habits.forEach((h: Habit, idx: number) => {
			if (h.order == null) h.order = idx;
		});
		return v as HabitData;
	}

	async repairData() {
		this.data = this.migrate(this.data);
		await this.save();
	}

    async importFromRaw(raw: any) {
		const data = this.migrate(raw);
		this.data = data;
		await this.save();
	}
}

import { Habit, HabitEval } from "./types";
import { evalHabitOnDate } from "./entries";

export type WeekdayKey = "Sun" | "Mon" | "Tue" | "Wed" | "Thu" | "Fri" | "Sat";

type ScorePoint = { date: string; score: number; };

// --- Exponential Smoothing (Loop Algorithm) ---
export function computeScoreTimeline(habit: Habit): ScorePoint[] {
	const entries = habit.entries;
	const entryDates = Object.keys(entries);
	let start = todayString();
	if (entryDates.length > 0) {
		entryDates.sort(compareDateStr);
		start = entryDates[0];
	}
	
	if (habit.createdAt) {
		const created = habit.createdAt.slice(0, 10);
		if (compareDateStr(created, start) < 0) start = created;
	}

	const today = todayString();
	if (compareDateStr(start, today) > 0) return [];

	const frequency = 1;
	const multiplier = Math.pow(0.5, Math.sqrt(frequency) / 13.0);
	let score = 0; 
	const result: ScorePoint[] = [];

	for (let d = start; compareDateStr(d, today) <= 0; d = addDays(d, 1)) {
		const ev = evalHabitOnDate(habit, d);
		let checkValue: number | null = null;

		if (ev === "OK") checkValue = 1;
		else if (ev === "NO") checkValue = 0;
		else checkValue = null;

		if (checkValue !== null) score = score * multiplier + checkValue * (1 - multiplier);
		else score = score * multiplier; 

		result.push({ date: d, score });
	}
	return result;
}

export function getOverallStats(habit: Habit): RangeStats {
	const today = todayString();
	const dates = Object.keys(habit.entries);
	let ok = 0; let total = 0;
	
	for (const d of dates) {
		if (compareDateStr(d, today) > 0) continue;
		const ev = evalHabitOnDate(habit, d);
		if (ev === "OK") { ok++; total++; }
		else if (ev === "NO") { total++; }
	}

	const timeline = computeScoreTimeline(habit);
	const last = timeline.length > 0 ? timeline[timeline.length - 1] : null;
	const percent = last ? Math.round(last.score * 100) : 0;

	return { ok, total, percent };
}

export function getTrendSummary(habit: Habit): TrendSummary {
	const timeline = computeScoreTimeline(habit);
	if (timeline.length === 0) return { monthDelta: 0, yearDelta: 0, monthCurrent: 0, yearCurrent: 0 };

	const today = todayString();
	const currentScore = timeline[timeline.length - 1].score * 100;

	const getScoreAt = (target: string): number => {
		if (compareDateStr(target, timeline[0].date) < 0) return 0;
		for (let i = 0; i < timeline.length; i++) {
			if (compareDateStr(timeline[i].date, target) >= 0) return timeline[i].score * 100;
		}
		return timeline[timeline.length - 1].score * 100;
	};

	const monthAgo = getScoreAt(addDays(today, -30));
	const yearAgo = getScoreAt(addDays(today, -365));

	return {
		monthDelta: Math.round(currentScore - monthAgo),
		yearDelta: Math.round(currentScore - yearAgo),
		monthCurrent: Math.round(currentScore),
		yearCurrent: Math.round(currentScore)
	};
}

export function buildHeatmapData(habit: Habit): HeatmapData {
	const map: HeatmapData = {};
	const today = todayString();
	const dates = Object.keys(habit.entries).sort();
	const start = dates.length > 0 ? dates[0] : today;

	for (let d = start; compareDateStr(d, today) <= 0; d = addDays(d, 1)) {
		const ev = evalHabitOnDate(habit, d);
		if (ev !== "NONE") map[d] = ev;
	}
	return map;
}

export function buildScoreSeries(habit: Habit, mode: "week" | "month" | "year"): LineSeries {
	const timeline = computeScoreTimeline(habit);
	const { from, to } = getRangeBounds(mode);
	const labels: string[] = []; const values: number[] = [];

	for (const p of timeline) {
		if (compareDateStr(p.date, from) >= 0 && compareDateStr(p.date, to) <= 0) {
			labels.push(p.date.slice(5));
			values.push(Math.round(p.score * 100));
		}
	}
	return { labels, values };
}

export function buildHistorySeries(habit: Habit, mode: "week" | "month" | "year"): LineSeries {
	const { from, to } = getRangeBounds(mode);
	const labels: string[] = []; const values: number[] = [];

	if (mode !== "year") {
		for (let d = from; compareDateStr(d, to) <= 0; d = addDays(d, 1)) {
			const ev = evalHabitOnDate(habit, d);
            const ent = habit.entries[d];
			labels.push(d.slice(5));
			values.push(habit.type === "yesno" ? (ev === "OK" ? 1 : 0) : (ent && typeof ent.value==='number' ? ent.value : 0));
		}
	} else {
        const monthMap = new Map<string, number>();
        const countMap = new Map<string, number>();
        const months = ["Ene","Feb","Mar","Abr","May","Jun","Jul","Ago","Sep","Oct","Nov","Dic"];
        
        for (let d = from; compareDateStr(d, to) <= 0; d = addDays(d, 1)) {
            const mIdx = parseInt(d.split('-')[1]) - 1;
            const key = months[mIdx];
            const ev = evalHabitOnDate(habit, d);
            const ent = habit.entries[d];
            
            if(!monthMap.has(key)) { monthMap.set(key, 0); countMap.set(key, 0); }
            
            if(habit.type === 'yesno') {
                if(ev === "OK") monthMap.set(key, monthMap.get(key)! + 1);
            } else {
                if(ent && typeof ent.value === 'number') {
                    monthMap.set(key, monthMap.get(key)! + ent.value);
                    countMap.set(key, countMap.get(key)! + 1);
                }
            }
        }
        monthMap.forEach((val, key) => {
            labels.push(key);
            if(habit.type === 'quant' && countMap.get(key)! > 0) values.push(Math.round(val / countMap.get(key)!));
            else values.push(val);
        });
	}
	return { labels, values };
}

// Wrappers para integración
export function getStreakInfoForHabit(storage: HabitStorage, habitId: string): StreakInfo {
    const habit = storage.getData().habits.find((h) => h.id === habitId);
    if(!habit) return {bestStreak:0, currentStreak:0};
    
    let curr = 0, best = 0;
    const today = todayString();
    const dates = Object.keys(habit.entries).sort();
    if(dates.length === 0) return {bestStreak:0, currentStreak:0};
    
    let cursor = dates[0];
    while(compareDateStr(cursor, today) <= 0) {
        if(evalHabitOnDate(habit, cursor) === "OK") {
            curr++; if(curr>best) best=curr;
        } else {
            // Revisar si es un día que rompe racha (NO). Si es NONE (no hecho, pero quizás no tocaba), depende de la lógica.
            // Para simplificar Loop: un día sin hacer rompe racha si se espera diario.
            if(evalHabitOnDate(habit, cursor) === "NO") curr = 0; 
        }
        cursor = addDays(cursor, 1);
    }
    return {bestStreak: best, currentStreak: curr, lastDate: dates[dates.length-1]};
}

export function getWeekdayStatsForHabit(storage: HabitStorage, habitId: string): Record<WeekdayKey, WeekdayStats> {
    const habit = storage.getData().habits.find((h) => h.id === habitId);
    const stats: any = {};
    ["Mon","Tue","Wed","Thu","Fri","Sat","Sun"].forEach(k => stats[k] = {ok:0, total:0, percent:0});
    if(habit) {
        Object.keys(habit.entries).forEach(d => {
            if(evalHabitOnDate(habit, d) !== "NONE") {
                const wd = weekdayShort(d);
                if(stats[wd]) {
                    stats[wd].total++;
                    if(evalHabitOnDate(habit, d) === "OK") stats[wd].ok++;
                }
            }
        });
    }
    Object.keys(stats).forEach(k => stats[k].percent = stats[k].total ? Math.round(stats[k].ok/stats[k].total*100) : 0);
    return stats;
}

export function getMoodDistribution(habit: Habit, mode: "week" | "month" | "year"): MoodStats {
    const { from, to } = getRangeBounds(mode);
    const counts: Record<string, number> = { 
        "😫": 0, "😐": 0, "🙂": 0, "😄": 0, "🔥": 0 
    };
    
    // Recorrer entradas
    for (const date in habit.entries) {
        if (compareDateStr(date, from) >= 0 && compareDateStr(date, to) <= 0) {
            const entry = habit.entries[date];
            // Solo contar si hay un mood registrado
            if (entry.mood && counts.hasOwnProperty(entry.mood)) {
                counts[entry.mood]++;
            }
        }
    }

    // Filtrar solo los que tienen datos para no llenar el gráfico de ceros
    const labels: string[] = [];
    const data: number[] = [];

    // Orden específico: de triste a fuego
    const order = ["😫", "😐", "🙂", "😄", "🔥"];
    
    order.forEach(mood => {
        if (counts[mood] > 0) {
            labels.push(mood);
            data.push(counts[mood]);
        }
    });

    return { labels, data };
}

// Helper para obtener promedio de energía
export function getAverageEnergy(habit: Habit, mode: "week" | "month" | "year"): string {
    const { from, to } = getRangeBounds(mode);
    let total = 0;
    let count = 0;

    for (const date in habit.entries) {
        if (compareDateStr(date, from) >= 0 && compareDateStr(date, to) <= 0) {
            const entry = habit.entries[date];
            if (entry.energy) {
                total += entry.energy;
                count++;
            }
        }
    }
    
    if (count === 0) return "-";
    return (total / count).toFixed(1); // Ej: "4.2"
}

import { HabitStorage } from "./storage";
import { Habit, HabitType, HabitFrequency, HabitGoal, HabitSortMode } from "./types";

function randomHabitId(): string {
	return "h_" + Date.now().toString(36) + "_" + Math.random().toString(36).slice(2, 10);
}

export interface CreateHabitOptions {
	name: string;
	type: HabitType;
	color?: string;
	icon?: string;
	frequency: HabitFrequency;
	goal?: HabitGoal;
}

export async function createHabit(storage: HabitStorage, opts: CreateHabitOptions): Promise<Habit> {
	if (!opts.name.trim()) throw new Error("Nombre obligatorio");
	const habit: Habit = {
		id: randomHabitId(),
		name: opts.name.trim(),
		type: opts.type,
		color: opts.color ?? "#FF8888",
		icon: opts.icon,
		frequency: opts.frequency,
		goal: opts.goal,
		archived: false,
		entries: {},
		createdAt: new Date().toISOString(),
		order: storage.getData().habits.length
	};
	await storage.update(d => d.habits.push(habit));
	return habit;
}

export async function updateHabit(storage: HabitStorage, habit: Habit) {
    await storage.update((d) => {
        const idx = d.habits.findIndex((h) => h.id === habit.id);
        if (idx !== -1) d.habits[idx] = habit;
    });
}

export async function archiveHabit(storage: HabitStorage, id: string) {
	await storage.update((d) => {
		const h = d.habits.find((x) => x.id === id);
		if (h) h.archived = true;
	});
}

export async function restoreHabit(storage: HabitStorage, id: string) {
	await storage.update((d) => {
		const h = d.habits.find((x) => x.id === id);
		if (h) h.archived = false;
	});
}

export async function deleteHabit(storage: HabitStorage, id: string) {
	await storage.update((d) => {
		d.habits = d.habits.filter((h) => h.id !== id);
	});
}

export function sortHabits(habits: Habit[], mode: HabitSortMode): Habit[] {
	const arr = [...habits];
	if (mode === "alpha") return arr.sort((a, b) => a.name.localeCompare(b.name));
	if (mode === "color") return arr.sort((a, b) => a.color.localeCompare(b.color));
	return arr.sort((a, b) => (a.order ?? 0) - (b.order ?? 0));
}

export async function reorderHabits(storage: HabitStorage, sourceId: string, targetId: string) {
    await storage.update((d) => {
        const habits = d.habits;
        const fromIdx = habits.findIndex((h) => h.id === sourceId);
        const toIdx = habits.findIndex((h) => h.id === targetId);
        if (fromIdx === -1 || toIdx === -1) return;
        const [moved] = habits.splice(fromIdx, 1);
        habits.splice(toIdx, 0, moved);
        habits.forEach((h, idx) => h.order = idx);
    });
}

        energyCard.createDiv("ham-card-sub").setText("En días completados");

        moodCard.style.height = "160px"; // Altura fija para el donut

		histContainer.style.height = "200px";

                    "😫": "#ef5350", // Rojo
                    "😐": "#bdbdbd", // Gris
                    "🙂": "#66bb6a", // Verde
                    "😄": "#42a5f5", // Azul

                        labels: moodData.labels,

                            data: moodData.data,

							label: "Score",
							data: scoreData.values,

							tension: 0.3,
							pointRadius: 3,
							pointBackgroundColor: "var(--background-primary)",
							pointBorderColor: this.habit.color,
							pointBorderWidth: 2
						}]
					},
					options: {
						responsive: true,
						maintainAspectRatio: false,
						interaction: { intersect: false, mode: 'index' },
						scales: { 
							x: { grid: { display: false }, ticks: { maxTicksLimit: 8 } }, 
							y: { display: false, min: 0, max: 105 } 
						},
						plugins: { legend: { display: false } }
					}
				}));
			}

			// --- CHART 2: BAR HISTORY ---
			const histData = buildHistorySeries(this.habit, this.currentRange);
			const ctxHist = histCanvas.getContext("2d");
			if (ctxHist) {
				this.charts.push(new Chart(ctxHist, {
					type: "bar",
					data: {
						labels: histData.labels,
						datasets: [{
							label: "Valor",
							data: histData.values,
							backgroundColor: this.habit.color,
							borderRadius: 3
						}]
					},
					options: {
						responsive: true,
						maintainAspectRatio: false,
						scales: { 
							x: { grid: { display: false }, ticks: { maxTicksLimit: 8 } }, 
							y: { grid: { color: "var(--background-modifier-border)" }, beginAtZero: true } 
						},
						plugins: { legend: { display: false } }
					}
				}));
			}
		});
	}

	onClose() {
		this.charts.forEach(c => c.destroy());
		this.contentEl.empty();
	}

import { App, Modal, Setting, Notice } from "obsidian";
import { HabitStorage } from "../../core/storage";
import { Habit, HabitEntry } from "../../core/types";
import { setEntry, deleteEntry, evalHabitEntry } from "../../core/entries";
import { createHabitNote, NoteSettings } from "../../utils/notes";

interface EntryModalOptions {
	storage: HabitStorage;
	habit: Habit;
	date: string;
	entry?: HabitEntry;
	onSave: (res: any) => void;
	onDelete: () => void;
}

export class EntryModal extends Modal {
	private storage: HabitStorage;
	private habit: Habit;
	private date: string;
	private entry?: HabitEntry;
	private onSave: (res: any) => void;
	private onDelete: () => void;

	private yesNoValue: "✔" | "✖" = "✔";
	private quantValue: number = 0;
	private notes: string = "";
    
    // Contexto
    private energyValue: number = 3;
    private moodValue: string = "";

	constructor(app: App, opts: EntryModalOptions) {
		super(app);
		this.storage = opts.storage;
		this.habit = opts.habit;
		this.date = opts.date;
		this.entry = opts.entry;
		this.onSave = opts.onSave;
		this.onDelete = opts.onDelete;

		// Inicializar valores
		if (this.habit.type === "yesno") {
			this.yesNoValue = this.entry?.value === "✖" ? "✖" : "✔";
		} else {
			this.quantValue = typeof this.entry?.value === "number" ? this.entry.value : 0;
		}
        
        this.energyValue = this.entry?.energy ?? 3;
        this.moodValue = this.entry?.mood ?? "";
	}

	onOpen(): void {
		const { contentEl } = this;
		contentEl.empty();
		contentEl.addClass("habit-entry-modal");

		contentEl.createEl("h2", { text: this.habit.name });
		contentEl.createDiv({ cls: "ht-entry-date", text: this.date });

		// Feedback visual
		const ev = evalHabitEntry(this.habit, this.entry);
		if (ev === "OK") contentEl.addClass("ht-entry-ok");
		else if (ev === "NO") contentEl.addClass("ht-entry-no");

		// --- Input Valor ---
		if (this.habit.type === "yesno") {
			const row = contentEl.createDiv("ht-entry-yesno-row");
			const btnOk = row.createEl("button", { text: "✔" });
			const btnNo = row.createEl("button", { text: "✖" });

			const update = () => {
				btnOk.toggleClass("is-active", this.yesNoValue === "✔");
				btnNo.toggleClass("is-active", this.yesNoValue === "✖");
			};
			btnOk.onclick = () => { this.yesNoValue = "✔"; update(); };
			btnNo.onclick = () => { this.yesNoValue = "✖"; update(); };
			update();
		} else {
			new Setting(contentEl).setName("Valor")
				.setDesc(this.habit.goal ? `Meta: ${this.habit.goal.value}` : "")
				.addText(t => {
					t.inputEl.type = "number";
					t.setValue(String(this.quantValue))
					 .onChange(v => this.quantValue = Number(v));
				});
		}

        // --- Contexto ---
        contentEl.createEl("h4", { text: "Contexto", cls: "ham-section-title" });
        
        const energyContainer = contentEl.createDiv("ht-context-row");
        new Setting(energyContainer)
            .setName("Nivel de Energía")
            .setDesc("1 (Baja) - 5 (Alta)")
            .addSlider(s => s
                .setLimits(1, 5, 1)
                .setValue(this.energyValue)
                .setDynamicTooltip()
                .onChange(v => this.energyValue = v)
            );

        const moodContainer = contentEl.createDiv("ht-mood-selector");
        const moods = ["😫", "😐", "🙂", "😄", "🔥"];
        moods.forEach(m => {
            const btn = moodContainer.createEl("button", { text: m, cls: "ht-mood-btn" });
            if (this.moodValue === m) btn.addClass("is-selected");
            btn.onclick = () => {
                this.moodValue = m;
                moodContainer.querySelectorAll(".ht-mood-btn").forEach(b => b.removeClass("is-selected"));
                btn.addClass("is-selected");
            };
        });

		// --- Notas ---
		new Setting(contentEl).setName("Nota del día")
			.addTextArea(t => {
				t.setPlaceholder("Opcional...")
                 .setValue(this.entry?.notePath ? "(Nota existente vinculada)" : "")
				 .onChange(v => this.notes = v);
			});

		// --- Footer ---
		const footer = contentEl.createDiv("habit-modal-footer");
		
		if (this.entry) {
			const btnDel = footer.createEl("button", { text: "Borrar Entrada", cls: "mod-warning" });
			btnDel.onclick = async () => {
				await deleteEntry(this.storage, this.habit.id, this.date);
				this.onDelete();
				this.close();
			};
		}

		const btnSave = footer.createEl("button", { text: "Guardar", cls: "mod-cta" });
		btnSave.onclick = async () => {
			const val = this.habit.type === "yesno" ? this.yesNoValue : this.quantValue;
			const s = this.storage.getData().settingsSnapshot;
			let notePath = this.entry?.notePath;

			// Mapeo correcto de tipos
			const noteSettings: NoteSettings = {
				folder: s.notesFolder,
				filenamePattern: s.noteFilenamePattern,
				template: s.noteTemplate,
				openAfterCreate: s.openNoteAfterCreate, // Mapeo correcto
				askBeforeCreate: s.askBeforeCreateNote  // Mapeo correcto
			};

            // CORRECCIÓN AQUÍ: Usar s.openNoteAfterCreate
			if (this.notes.trim() || s.openNoteAfterCreate) {
				if (s.askBeforeCreateNote) {
					if (confirm("¿Crear nota para este registro?")) {
						notePath = await createHabitNote(this.app, noteSettings, this.habit, this.date, String(val), this.notes);
					}
				} else {
					notePath = await createHabitNote(this.app, noteSettings, this.habit, this.date, String(val), this.notes);
				}
			}

			await setEntry(this.storage, this.habit.id, this.date, val, notePath, {
                energy: this.energyValue,
                mood: this.moodValue
            });
            
			this.onSave({ value: val, notePath });
			this.close();
		};
	}

	onClose() { this.contentEl.empty(); }
}

import { ItemView, WorkspaceLeaf, Menu, TFile, setIcon } from "obsidian";
import { HabitStorage } from "../../core/storage";
import { Habit, HabitEntry } from "../../core/types";
import { evalHabitOnDate } from "../../core/entries";
import { sortHabits, reorderHabits, deleteHabit, archiveHabit } from "../../core/habits";
import { getEntryForDate } from "../../core/entries"; 
import { todayString, addDays, weekdayShort } from "../../utils/dates";

// Importar Modales
import { CreateHabitModal } from "../modals/CreateHabitModal";
import { EditHabitModal } from "../modals/EditHabitModal";
import { EntryModal } from "../modals/EntryModal";
import { HabitAnalyticsModal } from "../modals/HabitAnalyticsModal";

export const HABIT_VIEW_TYPE = "habit-tracker-view";

export class HomeView extends ItemView {
	storage: HabitStorage;
	private detachEvents?: () => void;
    private draggingId: string | null = null;

	constructor(leaf: WorkspaceLeaf, storage: HabitStorage) {
		super(leaf);
		this.storage = storage;
	}

	getViewType(): string { return HABIT_VIEW_TYPE; }
	getDisplayText(): string { return "Habit Tracker"; }
	getIcon(): string { return "check-circle-2"; }

	async onOpen(): Promise<void> {
		this.render();
		const handler = () => this.render();
		// Escuchar eventos usando la API de Obsidian
		this.storage.events.on("changed", handler);
		this.detachEvents = () => {
			this.storage.events.off("changed", handler);
		};
	}

	async onClose(): Promise<void> {
		if (this.detachEvents) this.detachEvents();
	}

	render(): void {
		const container = this.containerEl;
		container.empty();
		container.addClass("habit-tracker-root");

		const data = this.storage.getData();
		const settings = data.settingsSnapshot;
        const today = todayString();

		// --- HEADER ---
		const header = container.createDiv("habit-tracker-header");
		header.createEl("h2", { text: "Habit Loop", attr: { style: "margin:0;" } });
        
        const controls = header.createDiv("ht-header-right");

        // Botón Sort
        const sortBtn = controls.createEl("button", { cls: "ht-btn", text: `Sort: ${settings.sortMode}` });
        sortBtn.onclick = async () => {
             const modes = ["manual", "alpha", "color"];
             const next = modes[(modes.indexOf(settings.sortMode) + 1) % modes.length] as any;
             await this.storage.update(d => d.settingsSnapshot.sortMode = next);
        };

        // Botón Nuevo
		const addBtn = controls.createEl("button", { cls: "ht-btn mod-cta", text: "+ Hábito" });
		addBtn.onclick = () => new CreateHabitModal(this.app, this.storage).open();

		// --- PREPARAR DATOS ---
		// 1. Filtrar
        let habits = data.habits.filter(h => !h.archived);
        if (settings.autoHideCompletedToday) {
            habits = habits.filter(h => evalHabitOnDate(h, today) !== "OK");
        }
        // 2. Ordenar
        habits = sortHabits(habits, settings.sortMode);

        // 3. Fechas (21 días)
		const dates: string[] = [];
        let base = today;
		for (let i = 20; i >= 0; i--) dates.push(addDays(base, -i));
		if (settings.dayBarOrientation === "recent-left") dates.reverse();

        // --- ESTRUCTURA TABLA ---
		const main = container.createDiv("ht-main");
		const leftCol = main.createDiv("ht-main-left");
		const rightColWrap = main.createDiv("ht-main-right");

		leftCol.createDiv("ht-left-header-spacer").setText("Hábito");

		const scrollStrip = rightColWrap.createDiv("ht-scroll-strip");
        
		const dayBar = scrollStrip.createDiv("habit-tracker-day-bar");
		dates.forEach(date => {
			const cell = dayBar.createDiv("ht-day-cell");
			cell.createDiv("ht-day-weekday").setText(weekdayShort(date, settings.firstDayOfWeek));
			cell.createDiv("ht-day-date").setText(date.slice(5)); // MM-DD
            if(date === today) cell.addClass("is-today");
		});

        if (habits.length === 0) {
            const empty = main.createDiv("ht-empty");
            empty.setText("No hay hábitos visibles.");
            empty.style.padding = "20px";
            empty.style.color = "var(--text-muted)";
            return;
        }

		habits.forEach(habit => {
            this.renderHabitRow(leftCol, scrollStrip, habit, dates, settings.sortMode);
		});
	}

    private renderHabitRow(leftCol: HTMLElement, scrollStrip: HTMLElement, habit: Habit, dates: string[], sortMode: string) {
        // 1. META (Columna Izq)
        const meta = leftCol.createDiv("ht-habit-meta");
        
        // Drag & Drop Logic
        if (sortMode === "manual") {
            meta.setAttr("draggable", "true");
            meta.addClass("ht-draggable");
            
            meta.ondragstart = (e) => { 
                this.draggingId = habit.id; 
                meta.addClass("is-dragging");
                e.dataTransfer?.setData("text/plain", habit.id);
            };
            meta.ondragend = () => { 
                this.draggingId = null; 
                meta.removeClass("is-dragging"); 
            };
            meta.ondragover = (e) => { e.preventDefault(); }; 
            meta.ondrop = async (e) => {
                e.preventDefault();
                if (this.draggingId && this.draggingId !== habit.id) {
                    await reorderHabits(this.storage, this.draggingId, habit.id);
                }
            };
        }

        // Contenido Visual
        meta.createDiv({ cls: "ht-habit-color", attr: { style: `background-color:${habit.color}` } });
        if(habit.icon) meta.createSpan({cls: "ht-habit-icon", text: habit.icon});
        meta.createSpan({cls: "ht-habit-name", text: habit.name});

        // Eventos Click Meta
        meta.onclick = () => new HabitAnalyticsModal(this.app, {storage: this.storage, habit}).open();
        
        meta.oncontextmenu = (e) => {
            e.preventDefault();
            const menu = new Menu();
            menu.addItem(i => i.setTitle("Analíticas").setIcon("bar-chart").onClick(() => new HabitAnalyticsModal(this.app, {storage: this.storage, habit}).open()));
            menu.addItem(i => i.setTitle("Editar").setIcon("pencil").onClick(() => new EditHabitModal(this.app, this.storage, habit, () => this.render()).open()));
            menu.addSeparator();
            menu.addItem(i => i.setTitle("Archivar").setIcon("archive").onClick(() => archiveHabit(this.storage, habit.id)));
            menu.addItem(i => i.setTitle("Eliminar").setIcon("trash").setWarning(true).onClick(() => {
                if(confirm(`¿Eliminar ${habit.name}?`)) deleteHabit(this.storage, habit.id);
            }));
            menu.showAtMouseEvent(e);
        };

        // 2. CELDAS (Columna Der)
        const row = scrollStrip.createDiv("ht-habit-cells-row");
        
        dates.forEach(date => {
            const cell = row.createDiv("ht-habit-cell");
            const ev = evalHabitOnDate(habit, date);
            const entry = habit.entries[date];

            const content = cell.createDiv("ht-cell-content");

            if (ev === "OK") {
                content.addClass("is-done");
                // Usar setAttr para style seguro
                content.setAttr("style", `color: ${habit.color}`); 
                setIcon(content, "check");
            } else if (ev === "NO") {
                content.addClass("is-fail");
                setIcon(content, "x");
            } else if (entry && typeof entry.value === "number") {
                 content.addClass("is-done");
                 content.setAttr("style", `color: ${habit.color}`);
                 content.createSpan({cls: "ht-cell-value", text: String(entry.value)});
            } else {
                content.addClass("is-none");
            }

            if (entry?.notePath) {
                const noteIcon = cell.createSpan({cls:"ht-cell-note"});
                // Click en notita abre el archivo
                noteIcon.onclick = (e) => {
                    e.stopPropagation();
                    const file = this.app.vault.getAbstractFileByPath(entry.notePath!);
                    if (file instanceof TFile) this.app.workspace.getLeaf(true).openFile(file);
                };
            }

            // Click en celda abre EntryModal
            cell.onclick = () => this.handleCellClick(habit, date, entry);
        });
    }

    private handleCellClick(habit: Habit, date: string, entry?: HabitEntry) {
        // CORRECCIÓN: Ya no pasamos noteSettings, el EntryModal los obtiene internamente
        new EntryModal(this.app, {
            storage: this.storage,
            habit,
            date,
            entry,
            onSave: async (res) => {
                // Callbacks opcionales si quisieras actualizar algo extra
            },
            onDelete: async () => {
                // Manejado por el modal
            }
        }).open();
    }
}
